struc SwAPIInfo
 aisLength              dw 10d  ; Size of this structure, in bytes
 aisAPI                 dw ?    ; API identifier
 aisMajor               dw ?    ; Major version number
 aisMinor               dw ?    ; Minor version number
 aisSupport             dw ?    ; Support level (NetBIOS: 1-4)
ends                            ; = 10d bytes

        ; aisAPI
        API_NETBIOS     = 01h   ; NetBIOS
        API_8022        = 02h   ; 802.2
        API_TCPIP       = 03h   ; TCP/IP
        API_LANMAN      = 04h   ; LAN Manager named pipes
        API_IPX         = 05h   ; NetWare IPX
