struc MID
 midInfoLevel   dw 0            ; Information level (must be zero)
 midSerialNum   dd ?            ; Serial number
 midVolLabel    db 11 dup (?)   ; ASCII volume label (space-padded)
 midFileSysType db 8 dup (?)    ; File system type (e.g. "FAT16   ")
ends                            ; = 25d bytes
